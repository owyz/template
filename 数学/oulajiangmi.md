$\Large A^B \equiv\begin{equation}  
\left\{  
             \begin{array}{**lr**}  
             A^{B\%\phi(C)} & gcd(A,C)=1 \\  
             A^B & gcd(A,C)\not=1,B<\phi(C) \\  
             A^{B\%\phi(C)+\phi(C)} &  gcd(A,C)\not=1,B\ge\phi(C)
             \end{array}  
\right.  
\end{equation}\ (mod\ C)$
